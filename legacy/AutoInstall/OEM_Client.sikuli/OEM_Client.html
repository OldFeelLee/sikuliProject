def focus_app(appName):
    global g2client
    g2client = App(appName)
    g2client.focus()

def openApp_logIn(path,image):
    check=str(g2client)
    if check[1]=='-':
        App.open(path)
        while not exists(image):
            wait(1)
        type(image,"12345678"+Key.ENTER)
