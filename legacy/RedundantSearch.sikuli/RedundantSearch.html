import datetime

pane = find()
device = find(Pattern().similar(0.91))

def Time():
    global s
    today = datetime.datetime.now()
    s = str(today)

while True:
    dragDrop(device,pane)
    wait(10)
    click(Pattern().targetOffset(-17,0))
    wait(5)
    click()
    wait(10)
    f= open("c:/a/Redundant.txt",'a')
    if exists(Pattern().exact()):
        Time()
        f.write(s+"    complete!\n")

    else:
        Time()
        f.write(s+"    fail!\n")
    f.close()
    type(Key.ALT+"s")
    wait(1)
    type("r")
    type("a")
    wait(10)
