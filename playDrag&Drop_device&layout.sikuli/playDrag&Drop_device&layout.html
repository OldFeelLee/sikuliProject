try:

    fileLog.cnt = 0
    fileLog.status = None
    pane = find()
    layout = find(Pattern().similar(0.95).targetOffset(8,10))
    device = find(Pattern().similar(0.90).targetOffset(7,9))

    while True:

        fileLog.status = "device"
        dragDrop(device,pane)
        wait(5)
        fileLog.status = "device drag"
        if forFunc.for_click_click(Pattern().similar(0.96).targetOffset(-7,3),Pattern().targetOffset(-22,3),1) == True :
            fileLog.status = "layout"
            randomFunc.ranNumber(10,300)
            wait(randomFunc.ranTime)
            dragDrop(layout,pane)
            fileLog.status = "layout drag"

            wait(5)
            fileLog.status = "layt"
        if forFunc.for_click_click(Pattern().similar(0.96).targetOffset(-7,3),Pattern().targetOffset(-22,3),1) == True :
            fileLog.status = "pane"
            randomFunc.ranNumber(10,300)
            wait(randomFunc.ranTime)
            rightClick(pane)
            clickFunc.click_click(,)
            fileLog.status = "clear"

        fileLog.Log("PlayTab_device&layout_06.25-07.02",'a',errorConst.no_error)

except FindFailed:
